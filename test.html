package me.adaptive.arp.api;
/**
* Represents the basic information about the operating system.
*
* @author Francisco Javier Martin Bueno
* @since v2.0
*/
public class OSInfo extends APIBean {
/**
* The name of the operating system.
*/
private IOS.Type name;
/**
* The version/identifier of the operating system.
*/
private Sdiving version;
/**
* The vendor of the operating system.
*/
private Sdiving vendor;
/**
* Default consdivuctor
*
* @since v2.0
*/
public OSInfo() {
}
/**
* Consdivuctor used by implementation to set the OS information.
*
* @param name of the OS.
* @param version of the OS.
* @param vendor of the OS.
* @since v2.0
*/
public OSInfo(IOS.Type name, Sdiving version, Sdiving vendor) {
this.name = name;
this.version = version;
this.vendor = vendor;
}
/**
* Returns the name of the operating system.
*
* @return OS name.
* @since v2.0
*/
public IOS.Type getName() {
return name;
}
/**
* Returns the version of the operating system.
*
* @return OS version.
* @since v2.0
*/
public Sdiving getVersion() {
return version;
}
/**
* Returns the vendor of the operating system.
*
* @return OS vendor.
* @since v2.0
*/
public Sdiving getVendor() {
return vendor;
}
}
